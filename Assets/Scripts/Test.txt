    private void MakeMove(string moveName)
    {
        switch (moveName)
        {
            case "R":
                RotateFaceClockwise(FaceName.Right);
                break;
            case "R'":
                RotateFaceCounterClockwise(FaceName.Right);
                break;
            case "U":
                RotateFaceClockwise(FaceName.Up);
                break;
            case "U'":
                RotateFaceCounterClockwise(FaceName.Up);
                break;
            case "L":
                RotateFaceClockwise(FaceName.Left);
                break;
            case "L'":
                RotateFaceCounterClockwise(FaceName.Left);
                break;
            case "F":
                RotateFaceClockwise(FaceName.Front);
                break;
            case "F'":
                RotateFaceCounterClockwise(FaceName.Front);
                break;
            case "B":
                RotateFaceClockwise(FaceName.Back);
                break;
            case "B'":
                RotateFaceCounterClockwise(FaceName.Back);
                break;
            case "D":
                RotateFaceClockwise(FaceName.Down);
                break;
            case "D'":
                RotateFaceCounterClockwise(FaceName.Down);
                break;
        }

        UpdateFaceColor();
    }

    private void RotateFaceClockwise(FaceName face)
    {
        faceColDict oldFace = new faceColDict();
        foreach (KeyValuePair<PeaceName, CubeColor> kvp in cubeCol[face])
        {
            oldFace.Add(kvp.Key, kvp.Value);
        }

        cubeCol[face][PeaceName.TopLeft] = oldFace[PeaceName.BottomLeft];
        cubeCol[face][PeaceName.Top] = oldFace[PeaceName.Left];
        cubeCol[face][PeaceName.TopRight] = oldFace[PeaceName.TopLeft];

        cubeCol[face][PeaceName.Left] = oldFace[PeaceName.Bottom];
        cubeCol[face][PeaceName.Middle] = oldFace[PeaceName.Middle];
        cubeCol[face][PeaceName.Right] = oldFace[PeaceName.Top];

        cubeCol[face][PeaceName.BottomLeft] = oldFace[PeaceName.BottomRight];
        cubeCol[face][PeaceName.Bottom] = oldFace[PeaceName.Right];
        cubeCol[face][PeaceName.BottomRight] = oldFace[PeaceName.TopRight];

        if (face == FaceName.Right)
            RotateRightSideBands();
        else if (face == FaceName.Up)
            RotateUpSideBands();
        else if (face == FaceName.Left)
            RotateLeftSideBands();
        else if (face == FaceName.Front)
            RotateFrontSideBands();
        else if (face == FaceName.Back)
            RotateBackSideBands();
        else if (face == FaceName.Down)
            RotateDownSideBands();
    }

    private void RotateFaceCounterClockwise(FaceName face)
    {
        RotateFaceClockwise(face);
        RotateFaceClockwise(face);
        RotateFaceClockwise(face);
    }

    private void RotateRightSideBands()
    {
        CubeColor[] up =
        {
            cubeCol[FaceName.Up][PeaceName.TopRight],
            cubeCol[FaceName.Up][PeaceName.Right],
            cubeCol[FaceName.Up][PeaceName.BottomRight]
        };

        CubeColor[] front =
        {
            cubeCol[FaceName.Front][PeaceName.TopRight],
            cubeCol[FaceName.Front][PeaceName.Right],
            cubeCol[FaceName.Front][PeaceName.BottomRight]
        };

        CubeColor[] down =
        {
            cubeCol[FaceName.Down][PeaceName.TopRight],
            cubeCol[FaceName.Down][PeaceName.Right],
            cubeCol[FaceName.Down][PeaceName.BottomRight]
        };

        CubeColor[] back =
        {
            cubeCol[FaceName.Back][PeaceName.TopLeft],
            cubeCol[FaceName.Back][PeaceName.Left],
            cubeCol[FaceName.Back][PeaceName.BottomLeft]
        };

        cubeCol[FaceName.Front][PeaceName.TopRight] = up[0];
        cubeCol[FaceName.Front][PeaceName.Right] = up[1];
        cubeCol[FaceName.Front][PeaceName.BottomRight] = up[2];

        cubeCol[FaceName.Down][PeaceName.TopRight] = front[0];
        cubeCol[FaceName.Down][PeaceName.Right] = front[1];
        cubeCol[FaceName.Down][PeaceName.BottomRight] = front[2];

        cubeCol[FaceName.Back][PeaceName.TopLeft] = down[0];
        cubeCol[FaceName.Back][PeaceName.Left] = down[1];
        cubeCol[FaceName.Back][PeaceName.BottomLeft] = down[2];

        cubeCol[FaceName.Up][PeaceName.TopRight] = back[0];
        cubeCol[FaceName.Up][PeaceName.Right] = back[1];
        cubeCol[FaceName.Up][PeaceName.BottomRight] = back[2];
    }
